cmake_minimum_required(VERSION 3.25)                            -- минимальная версия cmake
project(vector)                                                 -- название проекта

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -lpthread")       -- флаги компиляции

add_library(vector SHARED vector.cpp)                           -- создание динамической библиотеки

add_executable(vectorProg main.cpp vector.cpp)                  -- создание исполняемого файла проекта
target_link_libraries(vectorProg vector)                        -- линковка библиотек для исполняемого файла
